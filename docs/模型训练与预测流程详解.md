```python
# src/data/data_service.py

def prepare_datasets(self, data_path: str | Path) -> Dict[str, Any]:
    """
    准备数据集 - 完整的 ETL 流程
    
    输入：CSV 文件路径
    输出：训练/验证/测试数据集 bundle
    """
    csv_path = Path(data_path)
    
    # Step 1: 加载数据 + 特征工程
    X_raw, y_raw, feature_df = load_and_engineer_features(
        csv_path, 
        target_col="YD15"  # 目标列：15 分钟功率
    )
    # X_raw shape: (n_samples, n_features)
    # y_raw shape: (n_samples,)
    
    # Step 2: 构建滑动窗口序列
    X_seq, y_seq = build_sequences(
        X_raw, 
        y_raw, 
        lookback=24  # 使用过去 24 个时间点预测下一个点
    )
    # X_seq shape: (n_samples-lookback, lookback, n_features)
    # y_seq shape: (n_samples-lookback,)
    
    # Step 3: 按时间切分训练/验证/测试集
    bundle = split_by_time(
        X_seq, 
        y_seq,
        train_ratio=0.7,  # 70% 训练
        val_ratio=0.15,   # 15% 验证
        test_ratio=0.15   # 15% 测试
    )
    
    return {
        "bundle": bundle,
        "feature_names": feature_df.columns.tolist(),
        "input_dim": X_seq.shape[-1],  # 输入维度 = n_features
        "lookback": self.lookback,
    }
```

### 2.3 滑动窗口构建详解

```python
# src/data/dataset_builder.py

def build_sequences(X: np.ndarray, y: np.ndarray, lookback: int = 24):
    """
    构建滑动窗口序列
    
    示例（lookback=3）:
    
    原始数据:
    X: [x0, x1, x2, x3, x4, x5]
    y: [y0, y1, y2, y3, y4, y5]
    
    构建后:
    seq_X: [[x0,x1,x2], [x1,x2,x3], [x2,x3,x4], [x3,x4,x5]]
    seq_y: [y3, y4, y5, y6]
    
    即：用过去 3 个点预测下一个点
    """
    seq_X = []
    seq_y = []
    
    for i in range(lookback, len(X)):
        # 取过去 lookback 个点作为输入
        seq_X.append(X[i - lookback:i])
        # 取下一个点作为目标
        seq_y.append(y[i])
    
    return np.asarray(seq_X), np.asarray(seq_y)
```

```python
# src/models/lstm_model.py

def train(self, X_train, y_train, X_val, y_val, **kwargs):
    epochs = kwargs.get("epochs", 20)
    batch_size = kwargs.get("batch_size", 64)
    lr = kwargs.get("lr", 1e-3)
    
    # 1. 创建 DataLoader
    train_ds = TensorDataset(
        torch.tensor(X_train, dtype=torch.float32),
        torch.tensor(y_train, dtype=torch.float32),
    )
    train_loader = DataLoader(train_ds, batch_size=batch_size, shuffle=True)
    
    # 2. 定义损失和优化器
    criterion = nn.MSELoss()
    optimizer = torch.optim.Adam(self.model.parameters(), lr=lr)
    
    # 3. 训练循环
    for epoch in range(1, epochs + 1):
        epoch_losses = []
        for xb, yb in train_loader:
            # 前向传播
            preds = self.model(xb)
            loss = criterion(preds, yb)
            
            # 反向传播
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            
            epoch_losses.append(loss.item())
        
        train_loss = np.mean(epoch_losses)
        
        # 验证集评估
        val_loss = self._eval_loss(X_val, y_val)
        
        # 进度回调
        if progress_callback:
            progress_callback({
                "model": "lstm",
                "epoch": epoch,
                "train_loss": train_loss,
                "val_loss": val_loss,
            })
```

```python
def train(self, X_train, y_train, X_val, y_val, **kwargs):
    # 3D -> 2D 展平（XGBoost 不支持序列输入）
    X_train_2d = self._flatten(X_train)  # (n, lookback, features) -> (n, lookback*features)

    if X_val is not None and y_val is not None:
        eval_set = [(self._flatten(X_val), y_val)]
    
    # 直接调用 fit
    self.model.fit(
        X_train_2d,
        y_train,
        eval_set=eval_set,
        verbose=False,
    )
```

### 3.3 数据聚合逻辑

```python
# src/services/forecaster_manager.py

def _aggregate(series: np.ndarray, mode: str, target_points: int) -> np.ndarray:
    """
    数据聚合 - 将高频数据聚合为低频
    
    Args:
        series: 原始序列（15 分钟粒度）
        mode: 聚合模式 (none/hourly/daily)
        target_points: 目标点数
    
    示例：
    - hourly: [0:00, 0:15, 0:30, 0:45] -> mean -> [0:00]
    - daily: 96 个 15 分钟点 -> mean -> 1 天
    """
    if mode == "none":
        values = series
    
    elif mode == "hourly":
        # 每 4 个 15 分钟点聚合为 1 小时
        usable = (series.size // 4) * 4
        values = series[-usable:].reshape(-1, 4).mean(axis=1)
    
    elif mode == "daily":
        # 每 96 个 15 分钟点聚合为 1 天
        usable = (series.size // 96) * 96
        values = series[-usable:].reshape(-1, 96).mean(axis=1)
    
    # 填充到目标长度
    if values.size >= target_points:
        return values[-target_points:]
    else:
        pad_value = float(values[-1]) if values.size else 0.0
        padded = np.pad(values, (0, target_points - values.size), constant_values=pad_value)
        return padded
```

```
│ ensemble = 0.4*lstm + 0.35*gru + 0.15*xgboost + 0.10*moirai
│ result: (96,)
```

## 六、评估指标计算

### 6.1 指标公式

```python
# src/models/model_service.py

def _metrics(y_true: np.ndarray, y_pred: np.ndarray) -> Dict[str, float]:
    # MAE: 平均绝对误差
    mae = np.mean(np.abs(y_true - y_pred))
    
    # RMSE: 均方根误差
    rmse = np.sqrt(np.mean((y_true - y_pred) ** 2))
    
    return {"mae": mae, "rmse": rmse}
```

### 6.2 指标含义

| 指标 | 公式 | 含义 | 单位 |
|------|------|------|------|
| MAE | $\frac{1}{n}\sum\|y-\hat{y}\|$ | 平均绝对偏差，直观反映误差 | 与 y 相同 |
| RMSE | $\sqrt{\frac{1}{n}\sum(y-\hat{y})^2}$ | 均方根误差，惩罚大偏差 | 与 y 相同 |

## 七、常见问题解答

### Q1: 训练时为什么需要验证集？

**A:** 验证集用于：
1. 早停（Early Stopping）：验证 loss 不再下降时停止
2. Stacking 元学习器训练：用验证集预测作为元特征
3. 超参数调优

### Q2: Stacking 融合的原理是什么？

**A:** Stacking 是两层集成：
- 第一层：各基模型（LSTM/GRU/XGBoost/Moirai）独立预测
- 第二层：Ridge 回归学习如何组合各模型的预测

```python
# 元特征构造
X_meta = [lstm_pred, gru_pred, xgboost_pred, moirai_pred]  # shape: (n_val, 4)

# 训练 Ridge 回归
ridge.fit(X_meta, y_val)

# 最终预测
ensemble_pred = ridge.predict(X_meta)
```

### Q3: 为什么不同尺度使用不同权重？

**A:** 不同模型擅长不同时间尺度：
- **短期（天）**：LSTM/GRU 擅长捕捉高频波动
- **中期（周）**：XGBoost 对气象因子敏感
- **长期（月）**：Moirai 大模型擅长长周期趋势

### Q4: 如何添加新模型？

**A:** 三步：
1. 在 `src/models/` 下创建新模型类，继承 `BaseForecastModel`
2. 在 `build_model_registry()` 中注册新模型
3. 在 `HORIZON_CONFIGS` 中配置权重
